# -*- coding: utf-8 -*-
"""
SBERT Stress Score Pipeline - Clean Script
Process: Load Data -> Rename Users -> Embed Text -> Score Stress
"""

import pandas as pd
import numpy as np
from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity
import os

# --- Step 1: Mount Drive ---
try:
    from google.colab import drive
    drive.mount('/content/drive')
    IN_COLAB = True
except:
    IN_COLAB = False
    print("Running locally (not in Colab). Ensure file paths are correct.")

# --- Step 2: Paths & Data Loading ---
# Adjust these paths if running locally vs in Colab
project_path = '/content/drive/MyDrive/mentalhealth/' if IN_COLAB else './'
stressed_file = os.path.join(project_path, 'Categorized_Mental_Health_Data.csv')
normal_file = os.path.join(project_path, 'Dummy_Mental_Health_Conversations.csv')

df_stress = pd.read_csv(stressed_file)
df_normal = pd.read_csv(normal_file)

# --- Step 3: Transform 'User' to 'User_n' ---
# Specifically handles the 'User 1' -> 'User_n1' request
df_stress['user'] = df_stress['user'].str.replace('User ', 'User_n', regex=False)
df_normal['user'] = df_normal['user'].str.replace('User ', 'User_n', regex=False)

# Add user_type labels
df_stress['user_type'] = 'stressed'
df_normal['user_type'] = 'normal'

# Ensure unique IDs for the normal dataset to prevent collisions
df_normal['user'] = df_normal['user'].apply(lambda x: f"normal_{x}")

# --- Step 4: Combine & Clean ---
df_combined = pd.concat([df_stress, df_normal], ignore_index=True)
df_combined.columns = df_combined.columns.str.strip()

# --- Step 5: SBERT Model Loading ---
# Using MiniLM for speed and efficiency
model = SentenceTransformer('all-MiniLM-L6-v2')

# Define Stress Anchors (Representative of high stress)
stress_anchors = [
    "I feel overwhelmed with too many responsibilities and pressure",
    "I feel anxious, worried all the time and cannot relax",
    "I feel mentally exhausted and unable to cope anymore",
    "I feel like I am failing and nothing is working",
    "I feel emotionally drained and stuck in my situation"
]
stress_emb = model.encode(stress_anchors)

# --- Step 6: Vector Similarity & Scoring ---
# Turn off progress bar to avoid generating messy "Widget" metadata in the file
input_embeddings = model.encode(df_combined['input'].fillna('').tolist(), show_progress_bar=False)

def compute_stress_score(row_emb, anchor_embs):
    # Calculate cosine similarity between user text and all stress anchors
    sims = cosine_similarity(row_emb.reshape(1, -1), anchor_embs)[0]
    # Use the MAX similarity (closest match to any stress anchor)
    return int(round(max(sims) * 100))

df_combined['sbert_stress_score'] = [
    compute_stress_score(input_embeddings[i], stress_emb)
    for i in range(len(df_combined))
]

# --- Step 7: Exporting Final Data ---
combined_output = os.path.join(project_path, "combined_mentalhealth_data_updated.csv")
df_combined.to_csv(combined_output, index=False)

print(f"✅ Success! File saved at: {combined_output}")
print(df_combined[['user', 'user_type', 'sbert_stress_score']].head())
